293 void fftw3d_forward(GridFunction3d& inReal, GridFunction3d& inImag,
294 DoubleVector3d& outReal, DoubleVector3d& outImag)
295 {
296  long i,j,k,p;
297  long oldindex1, oldindex2, oldindex3;
298 
299  this->LX = inReal.getXmax() - inReal.getXmin();
300  this->LY = inReal.getYmax() - inReal.getYmin();
301  this->LZ = inReal.getZmax() - inReal.getZmin();
302 
303 
304  if((nx != inReal.getXpanelCount())
305  || (ny != inReal.getYpanelCount())
306  || (nz != inReal.getZpanelCount()))
307  {
308  initialize(inReal.getXpanelCount(),
309  inReal.getYpanelCount(),
310  inReal.getZpanelCount(),LX,LY,LZ);
311  }
312 
313  // Capture and re-order input. This extraction
314  // ignores the periodically defined values at
315  // the outer edges of the computational domain.
316 
317  for(i=0; i < nx; i++)
318  {
319  for(j=0; j < ny; j++)
320  {
321  for(k=0; k < nz; k++)
322  {
323  p = k + nz*(j + i*ny);
324 
325  in[p][0] = inReal(i,j,k);
326  in[p][1] = inImag(i,j,k);
327  }}}
328 
329  fftw_execute(forwardplan);
330 
331  //reorder AND scale output
332 
333  double scalingfactor = sqrt(LX*LY*LZ)/(nx*ny*nz);
334 
335 
336  for(p=0; p < nx*ny*nz; p++)
337  {
338 
339  oldindex1 = (p/nz)/ny;
340  oldindex2 = (p/nz)%ny;
341  oldindex3 = p%nz;
342 
343  i = (oldindex1 + (nx/2) )%nx ;//NB: uses integer division of odd numbers
344  j = (oldindex2 + (ny/2) )%ny ;//NB: uses integer division of odd numbers
345  k = (oldindex3 + (nz/2) )%nz ;//NB: uses integer division of odd numbers
346 
347  outReal(i,j,k) = out[p][0]*scalingfactor;
348  outImag(i,j,k) = out[p][1]*scalingfactor;
349  }
350 }

368  this->LZ = outReal.getZmax() - outReal.getZmin();
369

371

379  }
380 
381  for(i=0; i < nx; i++)
382  {
383  for(j=0; j < ny; j++)
384  {
385  for(k=0; k < nz; k++)
386  {
387  newindex1 = (i - (nx/2) + nx )%nx ;//NB: uses integer division of odd numbers
388  newindex2 = (j - (ny/2) + ny )%ny ;//NB: uses integer division of odd numbers
389  newindex3 = (k - (nz/2) + nz )%nz ;//NB: uses integer division of odd numbers
390

392

394  in[p][1] = inImag( i, j, k);
395  }}}
396 
397  //transform
398 
399  fftw_execute(inverseplan);
400 
401  double scalefactor = 1.0/sqrt(LX*LY*LZ);
402 
403  for(p=0; p < nx*ny*nz; p++)
404  {
405  // p = k + nz*(j + i*ny) = k + nz*j + i*nz*ny
406  // => p/nz = j + i*ny
407  //
408 
409  i = (p/nz)/ny;
410  j = (p/nz)%ny;
411  k = p%nz;
412 
413  outReal(i,j,k) = out[p][0]*scalefactor;
414  outImag(i,j,k) = out[p][1]*scalefactor;
415  }
416

418 
419  outReal.enforcePeriodicity();
420  outImag.enforcePeriodicity();
